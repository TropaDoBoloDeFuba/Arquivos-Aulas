Questão 1 

Nome do Banco de Dados: lista_exercicios1b

A - SELECT ID_NF, ID_ITEM, COD_PROD, VALOR_UNIT FROM `produtos` WHERE DESCONTO IS NULL;

B - SELECT ID_NF, ID_ITEM, COD_PROD, VALOR_UNIT, DESCONTO, (VALOR_UNIT - (VALOR_UNIT * (DESCONTO / 100))) AS VALOR_VENDIDO FROM produtos WHERE DESCONTO > 0;

C - UPDATE produtos SET DESCONTO = 0 WHERE DESCONTO IS NULL;

D - SELECT ID_NF, ID_ITEM, COD_PROD, VALOR_UNIT, (QUANTIDADE * VALOR_UNIT) AS VALOR_TOTAL, 
    DESCONTO,
    VALOR_UNIT - (VALOR_UNIT * (DESCONTO/100)) AS VALOR_VENDIDO FROM produtos;

E - SELECT ID_NF, SUM(QUANTIDADE * VALOR_UNIT) AS VALOR_TOTAL
    FROM produtos
    GROUP BY ID_NF
    ORDER BY VALOR_TOTAL DESC;

F - SELECT ID_NF, SUM(VALOR_UNIT - (VALOR_UNIT * (DESCONTO / 100))) AS VALOR_VENDIDO
    FROM produtos
    GROUP BY ID_NF
    ORDER BY VALOR_VENDIDO DESC;

G - SELECT COD_PROD, SUM(QUANTIDADE) FROM produtos GROUP BY COD_PROD ORDER BY QUANTIDADE DESC LIMIT 1;

H - SELECT ID_NF, COD_PROD, SUM(QUANTIDADE) FROM produtos GROUP BY ID_NF, COD_PROD HAVING SUM(QUANTIDADE) > 10;

I - SELECT ID_NF, SUM(QUANTIDADE * VALOR_UNIT) AS VALOR_TOTAL 
    FROM produtos 
    GROUP BY ID_NF 
    HAVING SUM(QUANTIDADE * VALOR_UNIT) > 500 
    ORDER BY VALOR_TOTAL DESC;

J - SELECT COD_PROD, AVG(DESCONTO) AS MEDIA FROM produtos GROUP BY COD_PROD;

K - SELECT COD_PROD, MIN(DESCONTO) AS MENOR, MAX(DESCONTO) AS MAIOR, AVG(DESCONTO) AS MEDIA FROM produtos GROUP BY COD_PROD;

L - SELECT ID_NF, COUNT(COD_PROD) AS QTD_ITENS FROM produtos GROUP BY ID_NF HAVING COUNT(COD_PROD) > 3;
